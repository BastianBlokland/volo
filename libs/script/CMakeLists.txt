message(STATUS "> library: script")

  src/args.c
  src/binder.c
  src/diag.c
  src/doc.c
  src/enum.c
  src/format.c
  src/eval.c

  test/config.c
  test/test_binder.c
  test/test_doc.c
  test/test_format.c
  test/test_eval.c
  test/test_lex.c
  test/test_mem.c
  test/test_read.c
  test/test_val.c
  test/utils.c
  )

configure_debuggable(test_lib_script)
